local function formatNumber(n)
    if n >= 1e9 then return string.format("%.2fB", n/1e9)
    elseif n >= 1e6 then return string.format("%.2fM", n/1e6)    -- States system Very Bad didn't know wht to make in time :(
    elseif n >= 1e3 then return string.format("%.1fK", n/1e3)
    else return tostring(n) end
end

local function teleportToSky()
    local part = Instance.new("Part")
    part.Size = Vector3.new(20, 2, 20)
    part.Anchored = true
    part.Position = Vector3.new(0, 2000, 0)
    part.Parent = workspace
    LocalPlayer.Character:MoveTo(part.Position)
end

local function returnToGround()

end

                local before = LocalPlayer.leaderstats.Rebirths.Value
                ReplicatedStorage.rEvents.rebirthEvent:FireServer("rebirthRequest")
                wait(1)
                local after = LocalPlayer.leaderstats.Rebirths.Value

                    isRebirthing = true
                    wait(0.5)
                    isRebirthing = false

                    loop = false
                end
            end)

        end
    else
        loop = false
    end
end)

spawn(function()
    while true do
        wait(1)
        if LocalPlayer.leaderstats then
            local steps = LocalPlayer.leaderstats.Steps.Value
            local rebirths = LocalPlayer.leaderstats.Rebirths.Value
            local hoops = LocalPlayer.leaderstats:FindFirstChild("Hoops") and LocalPlayer.leaderstats.Hoops.Value or 0
            local races = LocalPlayer.leaderstats:FindFirstChild("Races") and LocalPlayer.leaderstats.Races.Value or 0
            
            for _, tab in pairs(window.Tabs) do
                for _, element in pairs(tab.Elements) do
                    if element.Type == "Label" then
                        if element.Text:match("Steps:") then
                            element.Text = "Steps: " .. formatNumber(steps)
                        elseif element.Text:match("Rebirths:") then
                            element.Text = "Rebirths: " .. formatNumber(rebirths)
                        elseif element.Text:match("Hoops:") then
                            element.Text = "Hoops: " .. formatNumber(hoops)
                        elseif element.Text:match("Races:") then
                            element.Text = "Races: " .. formatNumber(races)
                        end
                    end
                end
            end
        end
    end
end)
