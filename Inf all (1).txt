local function RGBButton(text,pos)
	local btn = Instance.new("TextButton", Main)
	btn.Size = UDim2.new(0.4,0,0,36)
	btn.Position = pos
	btn.Text = text
	btn.Font = Enum.Font.GothamBold
	btn.TextSize = 14
	btn.BackgroundColor3 = Color3.fromRGB(30,30,30)
	Instance.new("UICorner", btn)

	RunService.RenderStepped:Connect(function()
		btn.TextColor3 = Color3.fromHSV(hue,1,1)
	end)

	return btn
end

	local key = string.upper(string.gsub(Box.Text,"%s+",""))

	if key == NORMAL_KEY or key == PAID_KEY or key == GHOST_KEY then
		getgenv().KEY_TYPE = key

		TweenService:Create(Main,TweenInfo.new(0.1),{
			Position = Main.Position
		}):Play()
	end
end)
